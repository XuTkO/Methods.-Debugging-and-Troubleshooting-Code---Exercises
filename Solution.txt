import java.util.Scanner;

public class Demo {
    public static void main(String[] args) {
    Scanner scanner = new Scanner(System.in);

        int firstNumber = Integer.parseInt(scanner.nextLine());
        int secondNumber = Integer.parseInt(scanner.nextLine());
        int thirdNumber = Integer.parseInt(scanner.nextLine());

        int resultOneAndTwo = getMaxNumber(firstNumber,secondNumber);
        int resultTwoAndThree = getMaxNumber(resultOneAndTwo,thirdNumber);
        System.out.println(resultTwoAndThree);

    }

     static int getMaxNumber(int firstNumber, int secondNumber) {
         if (firstNumber>secondNumber){
             return firstNumber;
         }
         return secondNumber;
    }
}
