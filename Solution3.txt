import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        double sideOfCube = Double.parseDouble(scanner.nextLine());
        String whichSideToCalculate = scanner.nextLine();

        System.out.printf("%.2f",calculateParameter(sideOfCube,whichSideToCalculate));

    }

     static double calculateParameter(double sideOfCube, String whichSideToCalculate) {
         double lenght=0.0;
         if (whichSideToCalculate.equals("face")){
             lenght=Math.sqrt(2*(Math.pow(sideOfCube,2)));
         } else if (whichSideToCalculate.equals("space")){
             lenght=Math.sqrt(3*(Math.pow(sideOfCube,2)));
         } else if (whichSideToCalculate.equals("volume")){
             lenght=Math.pow(sideOfCube,3);
         } else if (whichSideToCalculate.equals("area")){
             lenght=Math.pow(sideOfCube,6);
         }
         return lenght;
    }
}
